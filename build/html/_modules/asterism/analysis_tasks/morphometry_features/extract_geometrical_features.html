[docs]def do_extract_geometric_features(prod_original,prod_mapped_to_circle,image_id,extract,gabor_prod=None,unsharp_prod=None):
    """
    Extracts geometric features using the :func:`get_cluster_geometric_morphometric_features`

    Args:
        prod_original  (:class:`asterism.analysis_processes.morphometric_features_input_products.MorphometryInputData`):
        prod_mapped_to_circle  (:class:`asterism.analysis_processes.morphometric_features_input_products.MorphometryInputData`):
        image_id (int): id of the image

    Returns:
        feat_list ( list of :class:`asterism.core.morphometry.features.MorphometricFeatures`]):
        list of MorphometricFeatures objects for the geometric group

    """


    feat_list=[]
    for p,flag in zip([prod_original,prod_mapped_to_circle,unsharp_prod],['ic','depr','unsh']):
        if p is None:
            cluster=None
        else:
            cluster=p.cluster

        feat_list.append(get_cluster_geometric_morphometric_features(cluster,group_name='geom',name_flag=flag))


    return feat_list
