    """
    Function to estimate the backgorund level in a given image.

        - If `sub_block_frac_size` is provided, the image is partitioned in N-blocks with size=sub_block_frac_size*image>size,
        and the block with the lowest flux is used to estimate the bkg level. Otherwise, the full image is used.
        The first


        -If `skewness_range` is provided the skewness is minimized using the function :function:`.minimize_skewness`,
        i.e. the minimum   skewness in the range |m-s1,m+s1| is chosen, where m is the mean of the data, and s1=std(data)*`sigma_range`



    Parameters
    ----------
    image : :class:`.asterism.core.image_manager.image.Image`
        the image used to estimate the bkg

    threshold :  Optional[float]

    r_th : Optional[float]
    n_std_th : Optional[float]
    min_size_ratio : Optional[float]
    sub_block_frac_size : Optional[float]
    resampling_size : Optional[int]
    plot : Optional[bool]
    verbose : Optional[bool]

    Returns
    -------
    bkg_th : float
    bkg_mode : float
    bkg_sig : float



    """

    print("|setting threshold")
    print("|image flux min",image.array.min(),"image flux max",image.array.max())

    if plot==True:
        import pylab as plt

    if sub_block_frac_size is not None:
        print ("|using sub blocks")
        flux_bkg=image.get_lowest_flux_array_from_sub_stamp(sub_block_frac_size)
    else:
        flux_bkg=image.valid_flatten


    if resampling_size is not None:
        if flux_bkg.size>resampling_size:
            flux_bkg=np.random.choice(flux_bkg,size=resampling_size,replace=False)

    flux_bkg=np.sort(flux_bkg)




    if skewness_range is not None:
        print("|-")
        print("|skewness minimization",flux_bkg.size)
        print("|flux bkg array  min,max",flux_bkg.min(),flux_bkg.max())
        bkg_lvl,bkg_sig,skew,skew_range_mim,skew_range_max=minimize_skewness(flux_bkg,skewness_range,min_size_ratio,grid_size=100)

    else:
        bkg_lvl,bkg_sig,skew,size=eval_skew_minimization_stats(flux_bkg)


        skew_range_mim=None
        skew_range_max=None

    th=bkg_lvl+n_std_th*bkg_sig

    print("|-")
    print ("|bkg_lvl",bkg_lvl)
    print ("|bkg_std",bkg_sig)
    print ("|bkg_th",th)




    if plot==True:
        f, axarr = plt.subplots(2, )
        test_plot(axarr,flux_bkg,bkg_lvl,bkg_sig=bkg_sig,th=th,skew_1=skew_range_mim,skew_2=skew_range_max,pre_skew_min_bkg_lvl=flux_bkg.mean())
        plt.show()





    return th,bkg_lvl,bkg_sig
