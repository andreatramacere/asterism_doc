"""
Module: denclue_deblending
========================
Overview
--------
This modules provides the implementation of the deblending based on the :class:`.DENCLUE` algorithm.
The basic idea is to partition a cluster according to the clusters of attractors found by the
:class:`.DENCLUE`.  The resulting sub-clusters are analyzed and accepted or rejected according to some
user-specified criteria, and a final deblended list of source clusters is returned.
The Task is implemented by the :class:`.DoDENCLUEDeblendingTask` class, and the Task algorithm is implemented by
the :py:func:`.do_denclue_deblending` top-level function. This function implements a two step algorithm, as shonw in the graph below:
    - the configuration of the :class:`.DENCLUE` algorithm, and the denlcue partitioning of the clusters,  done  by the :py:func:`do_denclue_of_cluster`,
    - the validations of the resulting sub-clusters performed by the :py:func:`.get_deblended_cluster_list`

[docs]def get_pixel_matching_gauss_laplace_lm(parent_cluster,h,local_maxima_xy,off_x,off_y):
    lm_pix_coords=np.copy(local_maxima_xy)
    for ID,lm in enumerate(local_maxima_xy):
        #print ('x,y ',lm[0],lm[1],lm[0]+off_x,lm[1]+off_y)
        d=dist_eval(parent_cluster.cartesian.coords,x_c=lm[0]+off_x,y_c=lm[1]+off_y,metric='euclidean')
        msk1=d<h
        if parent_cluster.flux[msk1].size>0:
            lm_id=np.argmax(parent_cluster.flux[msk1])
            #move back to cluster image coords, shift to cluster coords is done later
            #print ('f1',parent_cluster.flux[msk1].max())
            lm_pix_coords[ID]=parent_cluster.cartesian.coords[msk1][lm_id]-[off_x,off_y]
    return lm_pix_coords



[docs]def gauss_laplace_down_sampling(parent_cluster,h,verbose=False):
    """

    Parameters
    ----------
    parent_cluster
    h_frac

    Returns
    -------

    """
    bkg=parent_cluster.flux.min()
    image,off_x,off_y,masked=parent_cluster.get_cluster_image_array(bkg=bkg)

    h2=h

    h1=max(1,h2*0.1)
    h_grid=np.int(h*0.5)
    if h_grid<=1:
        h_grid=1

    blob_log_det=BlobDetectionGaussLaplace(min_sigma=h1,
                                           max_sigma=h2,
                                           overlap=0.5)

    local_maxima_xy,local_maxima_flux=blob_log_det.apply(image,coords_in_xy=True,verbose=verbose)
    print(local_maxima_xy)
    if local_maxima_xy is None:
        target_coordinates=None
        target_coordinates_w_array=None

    elif len(local_maxima_xy)>1:
        #print ("local maxima",local_maxima_xy,local_maxima_flux)
        local_maxima_xy=get_pixel_matching_gauss_laplace_lm(parent_cluster,h,local_maxima_xy,off_x,off_y)
        #print ("pixel matched local maxima",local_maxima_xy,local_maxima_flux)

        if verbose==True:
            print("| gauss laplace h",h)

        #Local maxima added to target point final list
        r=local_maxima_xy[:,1].astype(np.int16)
        c=local_maxima_xy[:,0].astype(np.int16)

        m=masked[r,c]==False
        r=r[m]
        c=c[m]

        x=c
        y=r

        #grid to extract target points with step given by kernel width
        step=max(1.0,np.int(h_grid))
        row_sel=np.arange(0,image.shape[0],step,dtype=np.int16)
        col_sel=np.arange(0,image.shape[1],step,dtype=np.int16)
        mgrid=np.zeros(image.shape, dtype=np.bool)
        for row in row_sel:
            mgrid[row][col_sel]=True


        #target points are extracted from grid vertices
        #and appended to targets from local maxima
        y_grid,x_grid=np.where(np.logical_not(masked)*mgrid)
        x_t=np.append(x,x_grid).astype(np.int16)
        y_t=np.append(y,y_grid).astype(np.int16)

        #fluxes are of target points are extracted
        target_coordinates_w_array=image[np.int16(y_t),np.int16(x_t)]

        #cluster image offset is added to target coordinates
        x_t=x_t+off_x
        y_t=y_t+off_y
        target_coordinates=np.column_stack((x_t.astype(np.int16),y_t.astype(np.int16)))

    else:
        target_coordinates=None
        target_coordinates_w_array=None


    return target_coordinates,target_coordinates_w_array




[docs]def do_denclue_of_cluster(parent_cluster,
                          min_size,
                          kernel,
                          eps,
                          h_frac,
                          attr_dbs_K,
                          attr_dbs_eps,
                          mask_unchanged,
                          k_table_size,
                          R_max_frac,
                          R_max_kern_th,
                          plot=False,
                          digitize_attractors=False,
                          gl_downsampling=False,
                          verbose=False):
    """
    This function sets-up the parameters for the :class:`.DENCLUE`  partitioning of the input clusters.
    The following features are


    Parameters
    ----------
    parent_cluster
    min_size
    kernel
    eps
    h_frac
    attr_dbs_K
    attr_dbs_eps
    mask_unchanged
    k_table_size
    R_max_frac
    R_max_kern_th
    plot
    digitize_attractors
    gl_downsampling

    Returns
    -------

    """



    print ("|debelnding for cluster",parent_cluster.ID, "r_cluter",parent_cluster.r_cluster,"r_max",parent_cluster.r_max)

    if parent_cluster.cl_len>min_size:

        h=h_frac*np.sqrt(parent_cluster.r_max**2+parent_cluster.r_cluster**2)
        if 'gauss' not in kernel:
            if h<1.0:
                h=1.0



        if R_max_frac is not None:
            R_max_method='fixed'
            R_max=R_max_frac*parent_cluster.r_cluster

        elif R_max_kern_th is not None:
            R_max_method='kern_rel_th'
            R_max=None
        else:
            R_max_method='auto'
            R_max=None

        if verbose==True:
            print ("|eps",eps)
            print ("|h",h)
        denclue_debl=DENCLUE(eps=eps,h=h,kernel=kernel,k_table_size=k_table_size,R_max_method=R_max_method,R_max_th_rel=R_max_kern_th,R_max=R_max)

        if gl_downsampling == True:
            print("|downsampling start")
            target_coordinates,target_coordinates_w_array=gauss_laplace_down_sampling(parent_cluster,h)
            print("|downsamplig done")
            if target_coordinates is None and target_coordinates_w_array is None:
                return  [parent_cluster]

        else:
            target_coordinates=None
            target_coordinates_w_array=None

        print("|denclue")
        denclue_debl.run(parent_cluster.cartesian.coords,
                             dbscan_eps=attr_dbs_eps,
                             dbscan_K=attr_dbs_K,
                             weight_array=parent_cluster.pts_weight,
                             mask_unchanged=mask_unchanged,
                             R_max=R_max,
                             make_convolved_image=False,
                             digitize_attractors=digitize_attractors,
                             target_coordinates=target_coordinates,
                             target_coordinates_w_array=target_coordinates_w_array,
                             verbose=verbose)
        print("|denclue done")
        if plot is True:

            denclue_plot(denclue_debl,parent_cluster,target_coordinates=target_coordinates,target_coordinates_w_array=target_coordinates_w_array)

        denclue_clusters=denclue_debl.clusters_list
    else:
        denclue_clusters=[parent_cluster]


    return denclue_clusters

                          children_min_frac_integ_flux=0.2,
                          children_min_frac_peak_flux=0.5,
                          children_compact_frac_size=0.05,
                          children_ext_frac_size=0.5,
                          min_size=9,
                          mask_unchanged=0.5,
                          k_table_size=None,
                          R_max_frac=1.0,
                          R_max_kern_th=None,
                          plot=False,
                          denclue_catalog=False,
                          verbose=False,
                          do_denclue_deblending=True,
                          out_catalog=False,
                          digitize_attractors=False):

    """

    This function implements the top-level algorithm for the Denclue deblending:

        - Each cluster in the `cluster_list` is partitioned by the `.do_denclue_of_cluster` function.
        - The resulting denclue sub-clusters are processed by the `.get_deblended_cluster_list` that validates the sub-clusters, and returns the final list of accepted subclusters.


    Parameters
    ----------
    clusters_list
    eps
    h_frac
    kernel
    gl_downsampling
    attr_dbs_eps
    attr_dbs_K
    children_min_frac_integ_flux
    children_min_frac_peak_flux
    children_min_frac_size
    min_size
    mask_unchanged
    k_table_size
    R_max_frac
    R_max_kern_th
    plot
    denclue_catalog
    verbose
    do_denclue_deblending
    out_catalog
    digitize_attractors

    Returns
    -------

    """



    final_deblended_cluster_list=[]



    CL_ID=0
    for parent_cluster in clusters_list:



        print("|-")
        denclue_clusters=do_denclue_of_cluster(parent_cluster,
                                               min_size,
                                               kernel,
                                               eps,
                                               h_frac,
                                               attr_dbs_K,
                                               attr_dbs_eps,
                                               mask_unchanged,
                                               k_table_size,
                                               R_max_frac=R_max_frac,
                                               R_max_kern_th=R_max_kern_th,
                                               plot=plot,
                                               digitize_attractors=digitize_attractors,
                                               gl_downsampling=gl_downsampling,
                                               verbose=verbose)

        deblended_clusters_list=get_deblended_cluster_list(parent_cluster,
                                       denclue_clusters,
                                       validate_children,
                                       children_min_frac_integ_flux,
                                       children_min_frac_peak_flux,
                                       children_compact_frac_size,
                                       children_ext_frac_size,
                                       offset_x=0,
                                       offset_y=0,
                                       verbose=verbose)

        print("|-")
        if plot==True:
            plot_final_clusters(parent_cluster,deblended_clusters_list)

        for deblended_cluster in deblended_clusters_list:
            deblended_cluster.ID=CL_ID
            final_deblended_cluster_list.append(deblended_cluster)
            CL_ID+=1




    print ("|N final clusters=",len(final_deblended_cluster_list))
    return final_deblended_cluster_list
