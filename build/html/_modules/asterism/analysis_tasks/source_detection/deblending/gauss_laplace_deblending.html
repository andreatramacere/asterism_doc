    #remove local maxima below the detection th
    local_maxima_flux_parent_cluster=local_maxima_flux*0.0

    for ID in xrange(local_maxima_flux.size):
        local_maxima_flux_parent_cluster[ID]=parent_cluster_image[local_maxima_xy[ID,1],local_maxima_xy[ID,0]]

    local_maxima_xy=local_maxima_xy[local_maxima_flux_parent_cluster>th]
    lm_maxima_flux=local_maxima_flux_parent_cluster[local_maxima_flux_parent_cluster>th]

    print ("local_maxima above the th",lm_maxima_flux.shape[0])

    x_local_maxima=local_maxima_xy[:,0]
    y_local_maxima=local_maxima_xy[:,1]








    cl_multiplicity=np.zeros(len(cluster_list),dtype=np.int16)
    #assigned_lm_ID=np.zeros(len(cluster_list),dtype=np.int16)
    list_ID=np.arange(len(cluster_list))

    for ID_cl,cluster in enumerate(cluster_list):
        print("->",cluster.ID)

        for ID_lm,lm in enumerate(local_maxima_xy):

            #lm_ID[ID_lm]=ID_lm
            if cluster.check_is_in_cluster(x_local_maxima[ID_lm],y_local_maxima[ID_lm]) :
                cl_multiplicity[ID_cl]+=1
            #if cl_multiplicity[ID_lm]<=1:
            #    assigned_lm_ID=[ID_cl]


    unique=cl_multiplicity==1
    multiple=cl_multiplicity>1

    lost=local_maxima_xy.shape[0]-unique.sum()-multiple.sum()

    selected_cluster_list_ID=list_ID[unique]

    check = unique.sum()==local_maxima_xy.shape[0]

    assigned=multiple.sum()+unique.sum()


    good_list=[]
    if check:
        for ID in selected_cluster_list_ID:
            good_list.append(cluster_list[ID])

    print ('assigned lm',assigned, ' multiple',multiple.sum(), ' lost',lost,' unique',unique.sum(),check)
    return good_list

            th0=th_rel*det_image.min()

            for cl in conn.clusters_list:
                print("ID conn",cl.ID)
            for cl in good_list:
                print("ID good",cl.ID)
            print ("|sub clusters=%d for th=%f and trials=%d"%(len(conn.clusters_list),th,trial))
