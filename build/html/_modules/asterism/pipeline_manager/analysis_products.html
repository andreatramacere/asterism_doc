[docs]    def write_to_file(self):
        self.set_file_name()
        if self.data.dtype.names is not None:
            header=[i for i in self.data.dtype.names]
            header=', '.join(header)
        else:
            header=None

        fmt=''
        for dt in self.data.dtype.descr:
            if 'f' in dt[1]:
                fmt+='%e '
            elif 'i' in dt[1]:
                fmt+='%d '

            elif 's' in dt[1] or 'S' in dt[1]:
                fmt+'%s'
            else:
                fmt+='%f'

        np.savetxt(self.file_name,self.data,delimiter=' ',header=header,fmt=fmt)
