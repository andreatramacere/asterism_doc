[docs]def histogram_kde(v,bins=100,bw=None):
    x_bins=np.linspace(v.min(),v.max(),bins+1)

    x=(x_bins[1:]+x_bins[:-1])*0.5
    pdf = stats.gaussian_kde(v,bw_method=bw)(x)
    return pdf,x_bins,x





[docs]def histogram(v=None,bins=None,normed=False):
    x=(bins[1:]+bins[:-1])*0.5
    return np.histogram(v,bins=bins,normed=normed),x
