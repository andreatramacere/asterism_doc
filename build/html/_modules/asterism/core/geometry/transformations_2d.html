[docs]def cartesian_to_polar_2d(x,y,x_c,y_c):
    theta=np.arctan2(y-y_c,x-x_c)
    r=np.sqrt((x-x_c)*(x-x_c)+(y-y_c)*(y-y_c))
    #print (theta)
    theta_rad=np.unwrap(theta)
    #print (theta)
    return r,theta_rad





[docs]def polar_to_cartesian_2d(r,theta_rad,x_c=0,y_c=0):
    x=r*np.cos(theta_rad)
    y=r*np.sin(theta_rad)
    return x+x_c,y+y_c
