[docs]def eval_Gini(flux_array):
    """
    The Gini factor :cite:`Lotz:2004` indicates the distribution of the light among pixels. A value of Gini equal to 1
    would indicate that  all the light is concentrated in one pixel, on the contrary, a lower value would indicates that
    the light is distributed more evenly amongst the pixels, with Gini factor equal to 0, if all the pixels have the s
    ame flux value.

    Parameters
    ----------
    flux_array : 1dim numpy array,

    Returns
    -------
    Gini factor : float


    .. bibliography:: references.bib
    """
    flux_sorted_ids=np.argsort(flux_array)
    n=flux_sorted_ids.size
    if n>1:
        ID=np.arange(n)
        ID=ID+1
        flux_sorted=flux_array[flux_sorted_ids]
        g_down=np.fabs(flux_sorted.mean())*n*(n-1)

        g_up=(2*ID -n -1)*np.fabs(flux_sorted)

        g=g_up.sum()/g_down
    else:
        g=1.0

    return g




[docs]def eval_M20(flux_array,x,y,x_c,y_c):
    """

    Parameters
    ----------
    flux_array
    x
    y
    x_c
    y_c

    Returns
    -------

    """
    flux_sorted_ids=np.argsort(flux_array)[::-1]
    flux_sorted=flux_array[flux_sorted_ids]
    f_tot=flux_sorted.sum()

    ID_max=flux_sorted.size
    ID=1
    while flux_sorted[0:ID].sum()/f_tot<0.2 and ID<ID_max:
        ID+=1

    ID_20=ID-1
    x=x[flux_sorted_ids]
    y=y[flux_sorted_ids]
    dist=(x-x_c)**2+(y-y_c)**2
    m_tot=(flux_sorted*dist).sum()
    m_20=(flux_sorted*dist)[0:ID_20].sum()/m_tot


    if m_20<=0.0:
        return -100
    else:
        return np.log10(m_20)





[docs]def eval_Concentration(r_20,r_80,r_50,r_90):
    """
    Evaluates the concentration parameter:

    .. math::
        C=5log(r_{80}/r_{20})


    where :math:`r_{80}` and :math:`r_{20}` are the radii of the circular discs containing respectively 80\% and 20\% of the total cluster image flux.



    Parameters
    ----------
    r_20 : Optional
    r_80 : Optional

    Returns
    -------
    Concentration parameter float

    """

    if r_20>0 and r_80>0:


        c1= 1.0*np.log10(r_80/r_20)

    else:

        c1= 0


    if r_50>0 and r_90>0:


        c2= 1.0*np.log10(r_90/r_50)

    else:

        c2= 0

    return c1,c2

        cluster object from whicht to extract the features

    surf_bright_profile : :class: `asterism.core.photometry.surface_brightness_profile.ImageBrigthnessRadialProfile`

    name_flag : str

    feat_names=['Gini','M20','conc_1','conc_2','clumpiness','r_Petrosian_to_r_max','asymm','r_20_to_r_max','r_80_to_r_max','r_Petrosian']
    feat_dt=[np.float32]*len(feat_names)


    m_feat= MorphometricFeatures.feature_factory(feat_names,

                                                group_name,
                                                name_flag=name_flag,
                                                null_value=-99.,
                                                failed_value=np.nan)

        m_feat.fill_null()
    else:
        try:
            feature_values=[]

            r_20,r_80,r_50,r_90,r_Petrosian=eval_r_Petrosian(surf_bright_profile.I,surf_bright_profile.r,surf_bright_profile.s,image_array=cluster_image,plot=False)


            feature_values.append(eval_Gini(cluster.flux))

            feature_values.append(eval_M20(cluster.flux,cluster.cartesian.x,cluster.cartesian.y,cluster.x_c,cluster.y_c))


            feature_values.extend(eval_Concentration(r_20,r_80,r_50,r_90))

            print(r_20,r_80,r_50,r_90,r_Petrosian)
            cluster_image_blurred=GaussianFilter(sigma=r_Petrosian*blurring_scale).apply(cluster_image)

            feature_values.append(eval_clumpiness(cluster_image,cluster_image_blurred))

            feature_values.append(r_Petrosian/cluster.r_max)

            feature_values.append(eval_asymmetry(cluster_image,image_centroid[0],image_centroid[1],r_Petrosian))

            feature_values.append(r_20/cluster.r_max)

            feature_values.append(r_80/cluster.r_max)

            feature_values.append(r_Petrosian)

        except Exception,e:
