    def _build_dtype(self,group_name,names,dtypes,name_flag=None):
        _list=[]
        for name,dt in zip(names,dtypes):
            f_name=group_name+'_'+name
            if name_flag is not None:
                f_name+='_'+name_flag
            _list.append((f_name,dt))

        return np.dtype(_list)
