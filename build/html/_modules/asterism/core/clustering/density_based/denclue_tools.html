[docs]def donwsample_cluster_target_attractors(cluster,step):

    bkg=cluster.flux.min()
    image,off_x,off_y,masked=cluster.get_cluster_image_array(bkg=bkg)

    row_sel=np.arange(0,image.shape[0],step,dtype=np.int16)
    col_sel=np.arange(0,image.shape[1],step,dtype=np.int16)

    mgrid=np.zeros(image.shape, dtype=np.bool)
    for row in row_sel:
        mgrid[row][col_sel]=True

    y_t,x_t=np.where(np.logical_not(masked)*mgrid)


    target_coordinates_w_array=image[np.int16(y_t),np.int16(x_t)]


    x_t=x_t+off_x
    y_t=y_t+off_y
    target_coordinates=np.column_stack((x_t.astype(np.int16),y_t.astype(np.int16)))

    return target_coordinates,target_coordinates_w_array



[docs]def mask_unchanged_attractors(th,attractors_position_array,position_array):
    dx=np.fabs(attractors_position_array[:,0]-position_array[:,0])
    dy=np.fabs(attractors_position_array[:,1]-position_array[:,1])

    d=dx*dx+dy*dy

    selected=d>(th)

[docs]def eval_kernel_table(h,R_max,K_table_size,kern_type='gauss'):

    u=np.linspace(0,R_max,K_table_size)/h
    k_table_x=u
    k_table_y= _eval_kernel(kern_type,np.ones(K_table_size,dtype=np.float64),np.float64(u),h,K_table_size)

[docs]def convolved_map(position_array,
                    kernel_type,
                    h,
                    weight_array=None,
                    verbose=False,
                    R_max=None):

    border=1
    x_p=position_array[:,0]
    y_p=position_array[:,1]


    off_set_x=x_p.min()-border
    off_set_y=y_p.min()-border
    rows=int(y_p.max()-y_p.min())+1+2*border
    cols=int(x_p.max()-x_p.min())+1+2*border



    image=np.zeros((rows,cols),dtype=np.float32)

    n_rows=position_array.shape[0]

    if weight_array is None:
        weight_array=np.ones(x_p.size,dtype=np.float64)

    f_min=0
    for row_id in xrange(n_rows):


        d,w,x,y=get_local_data(position_array,weight_array,x_p[row_id],y_p[row_id],R_max)

        f=_eval_kernel(kernel_type,w.astype(np.float64),d.astype(np.float64),h,d.size)
        image[y_p[row_id]-off_set_y,x_p[row_id]-off_set_x]=f.sum()
        f_min=min(f.sum(),f_min)
    image[image==0]=f_min
    return image,off_set_x,off_set_y

[docs]def get_local_data(position_array,weight_array,x_c,y_c,R_max):


    dist=distance.dist_eval(position_array,x_c=x_c,y_c=y_c,metric='euclidean')

    if R_max is not None:

        selected= dist<=R_max

        w=weight_array[selected]

        d=dist[selected]

        x=position_array[:,0][selected]
        y=position_array[:,1][selected]


        return d,w,x,y
    else:
        return dist,weight_array,position_array[:,0],position_array[:,1]









[docs]def find_attractor_position(position_array,
                           weight_array,
                           x_t,
                           y_t,
                           row_id,
                           kernel_type,
                           h,
                           eps,
                           R_max,
                           max_calls=1000,
                           k_table_x=None,
                           k_table_y=None,
                           k_table_size=0,
                           f_is_precomputed=False):










        dist_eps=eps
        calls=0
        f=0


        while dist_eps>=eps and calls<max_calls:
            d,w,x,y=get_local_data(position_array,weight_array,x_t,y_t,R_max)

            if w.size>1:
                x_t_1,y_t_1,f=_update_position(w,
                                        d/h,
                                        x,
                                        y,
                                        w.size,
                                        h,
                                        kernel_type,
                                        k_table_x,
                                        k_table_y,
                                        k_table_size,
                                        f_is_precomputed)


            else:
                x_t_1=x_t
                y_t_1=y_t




            dist_eps=sqrt((x_t_1-x_t)*(x_t_1-x_t)+(y_t_1-y_t)*(y_t_1-y_t))

            x_t=x_t_1
            y_t=y_t_1
            calls=calls+1
            #print(dist_eps)
        #print("final pos",d,x,x_t_1,y_t_1,calls,eps,dist_eps)
        if calls>=max_calls:
            print("!Warning in find attractor ID=%d, calls=%d, eps=%f, dist_eps=%f, x_t_1=%f, y_t_1=%f"%(row_id,calls,eps,dist_eps,x_t_1,y_t_1))

        return x_t_1,y_t_1,calls,f
